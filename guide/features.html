üåä Shadow DOM Support

Penetrates Shadow DOM boundaries to extract encapsulated elements from web components and modern frameworks.

// Shadow DOM traversal
function extractFromShadowRoot(element) {
  if (element.shadowRoot) {
    const shadowElements = element.shadowRoot.querySelectorAll('*');
    return Array.from(shadowElements).map(analyzeElement);
  }
  return [];
}

üñºÔ∏è Iframe Analysis

Crosses iframe boundaries to analyze embedded content and nested documents.

// Iframe content extraction
function extractFromIframes() {
  const iframes = document.querySelectorAll('iframe');
  iframes.forEach(iframe => {
    try {
      const iframeDoc = iframe.contentDocument;
      if (iframeDoc) {
        extractElementsFromDocument(iframeDoc);
      }
    } catch (e) {
      console.log('Cross-origin iframe detected');
    }
  });
}
